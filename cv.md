```
/**
 * @param {number[]} nums
 * @return {number}
 */
var findNumbers = function(nums) {
    let result =0;

    nums.forEach((num)=>{
        const digitCounts = findDigitsCount(num);
        if(digitCounts%2===0){
            result++;
        }
    })

    return result;
};

var findDigitsCount = function(num){
    let result = 0;
    for (let i = 0; num >= 1; i++) {
      result++;
      num /= 10;
    }
    return result;
}
```
